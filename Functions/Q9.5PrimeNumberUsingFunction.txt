
#include <stdio.h>
#include <stdlib.h>
#include<math.h>

int isPrime(int n);

int main()
{
    int n1,n2;
    printf("Enter the first number : ");
    scanf("%d",&n1);
    printf("enter the second number : ");
    scanf("%d",&n2);
    printf("The prime numbers between %d and %d are : ",n1,n2);
    for(int i = n1; i <= n2; i++)
    {
        if(isPrime(i))
        {
            printf("%d ",i);
        }
    }
    return 0;
}

int isPrime(int n)
{
    for(int i = 2; i <= n/2; i++)
    {
        if(n%i == 0)
        {
            return 0;
        }
    }
    return 1;
}
